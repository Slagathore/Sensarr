# Image processing — required by pystray (tray icon rendering).
# Optional at runtime since Task H: if the tray stack can't import, the app

# Process inspection and management (hard reset, Plex process detection)

# .env file loading

# System-tray icon (Windows and Linux; optional at runtime, see Pillow note).
# There are intentionally NO Windows-only packages in this file — the split
# is expressed with sys_platform markers like the python-xlib line below, so
# a Linux install never pulls Windows-only behavior and a Windows install
# is unchanged.
pystray==0.19.5

# Linux tray backend for pystray (X11/XWayland). Wayland-only desktops that
# want a native indicator need system PyGObject + an AppIndicator library
# instead — the in-app Linux diagnostics spell this out.
python-xlib==0.33 ; sys_platform == "linux"

# (pip show, 2026-07-14). Bump deliberately, not by float.

# Fuzzy string matching — fallback for title similarity when Ollama is unavailable

# Sun Valley ttk theme — dark mode for the desktop UI (optional at runtime;
# the app falls back to the stock ttk look if missing)
sv-ttk==2.6.1

# Recycle-bin deletes for the Maintenance tab (falls back to permanent
# unlink if missing)
send2trash==2.1.0

# Subtitle fetching for the Library tab (optional at runtime — the button
# explains how to install it if missing)
subliminal==2.6.0
